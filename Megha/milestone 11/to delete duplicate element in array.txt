#include <iostream>
#define MAX_SIZE 1000
using namespace std;
int main() {
    int arr[MAX_SIZE],freq[MAX_SIZE];
    int i,j,n,count;
    cin>>n;
   for(i=0;i<n;i++){
 cin>>arr[i];
 freq[i]=-1;
   }
     for(i=0;i<n;i++){   
 for(j=i+1;j<n;j++){
     if(arr[i]==arr[j]){
    count++;
freq[j]=0;
 }
     }
     if(freq[i]!=0){
         freq[i]=count;
     }
     }
for(i=0;i<n;i++){ 
    if(freq[i]!=0){
        cout<<arr[i]<<",";
    }
}
   return 0;
}
