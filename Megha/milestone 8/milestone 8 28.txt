#include <iostream>
using namespace std;
int main() {
    int k=0,m=1;
for(int i=1;i<=5;i++){
      
    for(int j=1;j<=9;j++){
    
        if(j<=i+k){
            if(j%2==0){
        cout<<"*  ";
            }
            else {
              cout<<m<<" ";
              m++;
            } 
        }
        else{
        cout<<"   ";
        }
         }
         k++;
         cout<<endl;
}
return 0;
}
