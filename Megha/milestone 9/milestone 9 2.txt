#include <iostream>
using namespace std;
int main() {
    int k;
  for(int i=5;i>=1;i--){
      k=i;
      for(int j=1;j<=5;j++){
          if(j<=6-i){
              cout<<k<<" ";
              k++;
          }
          else{
              cout<<"  ";
          }
      }
      cout<<endl;
  }
  return 0;
}
